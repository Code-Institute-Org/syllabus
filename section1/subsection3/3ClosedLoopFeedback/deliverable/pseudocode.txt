Control loop feedback, braitenberg example:

Controller {
  initialize_the_robot();
  threshold = x #set the value from which the robot must turn to avoid collision
  while(until_robot_is_turn_off) {
    distance_sensor_left = get_left_distance_sensor_value();
    distance_sensor_right = get_right_distance_sensor_value();
    if (distance_sensor_left > threshold)
      turn_right()
    else if (distance_sensor_right > threshold)
      turn_left()
    else
      move_forward
  }
}
